package com.escapethemansion.viewer;

import com.escapethemansion.gui.GUI;
import com.escapethemansion.model.*;
import com.escapethemansion.model.game.element.*;
import com.escapethemansion.model.game.floor.Floor;

import java.io.IOException;
import java.util.List;

public class FloorViewer {
    private final GUI gui;

    public FloorViewer(GUI gui){
        this.gui = gui;
    }

    public void draw(Floor floor) throws IOException {
        gui.clear();

        gui.drawText(new Position(11,5), floor.getTitle(), "#FFFFFF");

        drawElements(floor.getWalls(), new ElementViewer<Wall>());
        drawElements(floor.getGhosts(), new ElementViewer<Ghost>());
        drawElements(floor.getKeys(), new ElementViewer<Key>());
        drawElement(floor.getPlayer(), new ElementViewer<Player>());
        gui.refresh();
    }

    private <T extends Element> void drawElements(List<T> elements, ElementViewer<T> viewer){
        for (T element : elements){
            drawElement(element, viewer);
        }
    }

    private <T extends Element> void drawElement(T element, ElementViewer<T> viewer) {
        viewer.drawElement(this.gui, element);
    }

    public GUI.ACTION getNextAction() throws IOException {
        return gui.getNextAction();
    }

    public void close() throws IOException {
        gui.close();
    }
}
