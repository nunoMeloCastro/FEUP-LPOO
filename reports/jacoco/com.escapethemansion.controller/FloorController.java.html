package com.escapethemansion.controller;

import com.escapethemansion.gui.GUI;
import FloorViewer;

import com.escapethemansion.model.game.floor.Floor;

import java.io.IOException;
import java.util.Objects;

public class FloorController extends GameController{
    private final PlayerController playerController;
    private final GhostController ghostController;

    private final FloorViewer viewer;

    public FloorController(Floor floor, FloorViewer viewer) {
        super(floor);

        this.viewer = viewer;
        this.playerController = new PlayerController(floor);
        this.ghostController = new GhostController(floor);
    }

    public PlayerController getPlayerController(){
        return this.playerController;
    }

    public GhostController getGhostController() {
        return this.ghostController;
    }

    public void start() throws IOException {
        // TODO: ver qual a melhor condição de término do jogo
        while (true){
            viewer.draw(getFloor());

            GUI.ACTION action = viewer.getNextAction();
            if(action == GUI.ACTION.QUIT) break;

            playerController.doAction(action);
            ghostController.doAction();
        }

        viewer.close();
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        FloorController that = (FloorController) o;
        return Objects.equals(playerController, that.playerController) && Objects.equals(ghostController, that.ghostController) && Objects.equals(viewer, that.viewer);
    }

    @Override
    public int hashCode() {
        return Objects.hash(playerController, ghostController, viewer);
    }
}
