package com.escapethemansion.controller;

import com.escapethemansion.model.game.element.Ghost;
import com.escapethemansion.model.game.Position;
import com.escapethemansion.model.game.element.Key;
import com.escapethemansion.model.game.element.Wall;
import com.escapethemansion.model.game.floor.Floor;

import java.util.Objects;

public abstract class ElementController {

    private final Floor floor;

    public ElementController (Floor floor) {
        this.floor = floor;
    }

    public Floor getFloor() {
        return this.floor;
    }

    public abstract void moveElementUp();

    public abstract void moveElementRight();

    public abstract void moveElementDown();

    public abstract void moveElementLeft();

    public abstract void moveElement(Position position);

    public boolean canElementMove(Position position){
        for (Wall wall : this.getFloor().getWalls())
            if (wall.getPosition().equals(position)) return false;

        for (Ghost ghost : this.getFloor().getGhosts())
            if (ghost.getPosition().equals(position)) return false;

        for (Key key : this.getFloor().getKeys())
            if (key.getPosition().equals(position)) return false;

        return !this.getFloor().getPlayer().getPosition().equals(position);
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        ElementController that = (ElementController) o;
        return Objects.equals(floor, that.floor);
    }

    @Override
    public int hashCode() {
        return Objects.hash(floor);
    }
}
