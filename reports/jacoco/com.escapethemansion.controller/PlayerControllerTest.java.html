package com.escapethemansion.controller;

import com.escapethemansion.gui.GUI;
import com.escapethemansion.model.game.Position;
import com.escapethemansion.model.game.floor.Floor;
import com.escapethemansion.model.game.floor.FloorBuilder;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

import java.io.IOException;

public class PlayerControllerTest {

    private PlayerController playerController;

    @BeforeEach
    void setup() throws IOException {
        Floor floor = new FloorBuilder("src/test/resources/floors/floor.map").createFloor();
        this.playerController = new PlayerController(floor);
    }

    @Test
    void getFloor() throws IOException {
        Floor test = new FloorBuilder("src/test/resources/floors/floor.map").createFloor();
        assertEquals(this.playerController.getFloor(), test);
    }

    @Test
    void canPlayerMove() {
        assertFalse(this.playerController.canElementMove(new Position(18,21)));
        assertTrue(this.playerController.canElementMove(new Position(18,20)));
    }

    @Test
    void doAction() {
        GUI.ACTION action = GUI.ACTION.UP;
        this.playerController.doAction(action);
        assertEquals(this.playerController.getFloor().getPlayer().getPosition(), new Position(18, 20));

        action = GUI.ACTION.RIGHT;
        this.playerController.doAction(action);
        assertEquals(this.playerController.getFloor().getPlayer().getPosition(), new Position(19, 20));

        action = GUI.ACTION.DOWN;
        this.playerController.doAction(action);
        assertEquals(this.playerController.getFloor().getPlayer().getPosition(), new Position(19, 21));

        action = GUI.ACTION.LEFT;
        this.playerController.doAction(action);
        assertEquals(this.playerController.getFloor().getPlayer().getPosition(), new Position(18, 21));
    }
}
